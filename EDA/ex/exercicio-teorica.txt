invert(Queue Q){
    stack S = createstack(100);
    while(!IsEmpty(Q)){
        Push(Dequeue(Q), S);
    }
    makeEmpty(Q);
    while(!IsEmpty(S)){
        Enqueue(Pop(S), Q);
    }
    return Q;
}
